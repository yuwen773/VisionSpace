  let accumulatedContent = ''

  for (const chunk of stream) {
    if (chunk.node === 'heartbeat') continue

    if (chunk.error) {
      // 统一错误处理
      messages.value.push({ type: 'error', content: chunk.errorMessage })
      continue
    }

    if (chunk.chunk) {
      if (isFirstChunk) {
        // 创建新消息
        messages.value.push({ type: 'assistant', content: chunk.chunk })
        isFirstChunk = false
      } else {
        // 追加到末尾
        accumulatedContent += chunk.chunk
        const lastMsg = messages.value[messages.value.length - 1]
        lastMsg.content = accumulatedContent
      }
    }

    if (chunk.message) {
      // 添加工具消息为新消息
      messages.value.push({ type: chunk.message.messageType, ... })
      isFirstChunk = true  // 重置
      accumulatedContent = ''
    }
  }

是什么意思？                                                                    
● 这个警告的意思是：

  警告含义

  Spring MVC 在处理异步请求（SSE/流式响应）时，使用了默认的
  SimpleAsyncTaskExecutor，这个执行器不适合生产环境高负载使用。

  "这个执行器不适合生产环境使用，请在 WebMvc 配置中设置一个 AsyncTaskExecutor"

  为什么会有这个警告

  SSE 流式响应是异步处理，Spring MVC
  默认使用简单的异步执行器，没有线程池复用，在高并发下会有性能问题。

  是否需要修复

  ┌───────────┬────────────────┐
  │   环境    │      建议      │
  ├───────────┼────────────────┤
  │ 开发/测试 │ 可以忽略       │
  ├───────────┼────────────────┤
  │ 生产环境  │ 建议配置线程池 │
  └───────────┴────────────────┘

  修复方案

  在配置类中添加异步执行器配置：

  @Configuration
  public class WebMvcConfig implements WebMvcConfigurer {

      @Bean
      public AsyncTaskExecutor asyncTaskExecutor() {
          ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
          executor.setCorePoolSize(4);
          executor.setMaxPoolSize(16);
          executor.setQueueCapacity(100);
          executor.setThreadNamePrefix("async-");
          executor.initialize();
          return executor;
      }

      @Override
      public void configureAsyncSupport(AsyncSupportConfigurer configurer) {
          configurer.setTaskExecutor(asyncTaskExecutor());
      }
  }

  ┌───────────┬─────────────────────────┬─────────────────┐
